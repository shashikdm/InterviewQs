## Solution:
smartly use % and /
```
#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long t, n, k, steps;
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        steps = 0;
        if(k == 1)
        {
            cout<<n<<'\n';
            continue;
        }
        while(n > 0)
        {
            steps += n%k+1;
            n = n/k;
        }
        cout<<steps-1<<'\n';
    }
    return 0;
}
```
